Basic Game

A complete example of a Texas Hold'em poker hand with various agent types.

Features demonstrated:

  • Full game flow with betting rounds
  • Multiple agent types
  • Hand evaluation
  • Pot distribution

from pokersim.game.state import GameState
from pokersim.agents.random_agent import RandomAgent
from pokersim.agents.rule_based_agent import RuleBasedAgent

# Set up a game with 6 players
game_state = GameState(num_players=6)

# Create agents (mix of random and rule-based)
agents = [
    RandomAgent(0),
    RuleBasedAgent(1),
    RandomAgent(2),
    RuleBasedAgent(3),
    RandomAgent(4),
    RuleBasedAgent(5)
]

# Deal cards and start the game
game_state.deal_hole_cards()

# Play through the hand
while not game_state.is_terminal():
    current_player = game_state.current_player
    agent = agents[current_player]
    action = agent.act(game_state)
    game_state = game_state.apply_action(action)

# Determine winner and distribute pot
results = game_state.get_results()
print(f"Winner: Player {results['winners'][0]}")
print(f"Winning hand: {results['winning_hand']}")
print(f"Pot: {results['pot']}")

Spin & Go Tournament

Simulation of a complete Spin & Go tournament with prize multipliers and eliminations.

Features demonstrated:

  • Prize pool multipliers
  • Tournament progression
  • Player eliminations
  • Blind level increases

from pokersim.game.spingo import SpinGoGame
from pokersim.agents.advanced_opponent_agent import AdvancedOpponentAgent

# Create a Spin & Go tournament
tournament = SpinGoGame(
    buy_in=10,
    num_players=3,
    starting_stack=500
)

# Set up advanced agents
agents = [
    AdvancedOpponentAgent(0),
    AdvancedOpponentAgent(1),
    AdvancedOpponentAgent(2)
]

# Run the tournament until completion
while not tournament.is_tournament_over():
    # Start a new hand
    game_state = tournament.start_new_hand()
    
    # Play the hand
    while not game_state.is_terminal():
        current_player = game_state.current_player
        action = agents[current_player].act(game_state)
        game_state = game_state.apply_action(action)
    
    # Update player stacks after the hand
    tournament.update_stacks_after_hand(game_state)

# Get the winner
winner = tournament.get_winner()
print(f"Tournament winner: Player {winner}")
print(f"Prize: ${tournament.get_prize(winner)}")

Deep CFR Training

Training a poker agent using Deep Counterfactual Regret Minimization.

Features demonstrated:

  • Deep CFR algorithm implementation
  • Neural network integration
  • Agent evaluation
  • Performance metrics

from pokersim.algorithms.deep_cfr import DeepCFR
from pokersim.ml.models import ValueNetwork
import torch

# Create value networks
value_net = ValueNetwork(
    input_size=128,
    hidden_size=256,
    output_size=1
)

# Initialize Deep CFR algorithm
deep_cfr = DeepCFR(
    value_network=value_net,
    game_type='holdem',
    num_iterations=1000,
    memory_size=10000,
    batch_size=128,
    learning_rate=0.001
)

# Train the agent
deep_cfr.train()

# Evaluate against baseline agents
results = deep_cfr.evaluate(num_games=1000)
print(f"Win rate: {results['win_rate']:.2f}")
print(f"Average profit: {results['avg_profit']:.2f}")

Advanced Opponent Modeling

Creating sophisticated opponents with different playing styles and adaptability.

Features demonstrated:

  • Opponent profiles and playing styles
  • Adaptation to player behavior
  • Hand strength evaluation
  • Position-based strategy

from pokersim.agents.advanced_opponent_agent import (
    AdvancedOpponentAgent, 
    AdvancedOpponentProfile
)

# Create different opponent profiles
tight_aggressive = AdvancedOpponentProfile(
    aggression=0.8,
    bluff_frequency=0.1,
    call_threshold=0.6,
    raise_threshold=0.7
)

loose_passive = AdvancedOpponentProfile(
    aggression=0.3,
    bluff_frequency=0.4,
    call_threshold=0.3,
    raise_threshold=0.7
)

loose_aggressive = AdvancedOpponentProfile(
    aggression=0.9,
    bluff_frequency=0.3,
    call_threshold=0.3,
    raise_threshold=0.5
)

# Create agents with different profiles
agents = [
    AdvancedOpponentAgent(0, tight_aggressive),
    AdvancedOpponentAgent(1, loose_passive),
    AdvancedOpponentAgent(2, loose_aggressive)
]

# Use these agents in game simulations
game_state = GameState(num_players=3)
game_state.deal_hole_cards()

while not game_state.is_terminal():
    current_player = game_state.current_player
    action = agents[current_player].act(game_state)
    game_state = game_state.apply_action(action)
